    def get_announcement(s, url, secName):
    try:
        r = s.get(url)
        filename = str(secName) + ".pdf" 
         # 注意要用'wb', b表示二进制，不要用'w'
        with open(filename, 'wb') as f: 
             # r.content -> requests中的二进制响应内容：以字节的方式访问请求响应体
            f.write(r.content)
